    EYELINE-MSAG-MIB DEFINITIONS ::= BEGIN
    IMPORTS
        sibinco 
    FROM SIBINCO-SMSC-MIB
        DisplayString
    FROM SNMPv2-TC;

    -- sibinco is 1.3.6.1.4.1.26757
    -- sibinco is iso.org.dod.internet.private.enterprises.sibinco

    -- 1.3.6.1.4.1.26757.2
    msag OBJECT IDENTIFIER ::= { sibinco 2 }

    -- 1.3.6.1.4.1.26757.2.0
    msagnotifications OBJECT IDENTIFIER ::= { msag 0 }

    -- 1.3.6.1.4.1.26757.2.1
    msagDescr OBJECT-TYPE
    SYNTAX  DisplayString
    MAX-ACCESS  read-only
    STATUS  current
    DESCRIPTION
    "Version string"
    ::= { msag 1 }

    -- 1.3.6.1.4.1.26757.2.4
    msagCounterTable OBJECT-TYPE
    SYNTAX  SEQUENCE OF MsagCounterEntry
    MAX-ACCESS  not-accessible
    STATUS  current
    DESCRIPTION
    "MSAG Counter table."
    ::= { msag 4 }

    -- 1.3.6.1.4.1.26757.2.4.1
    msagCounterEntry OBJECT-TYPE
    SYNTAX  MsagCounterEntry
    MAX-ACCESS  not-accessible
    STATUS  current
    DESCRIPTION
    "MSAG Counter entry."
    INDEX { msagCounterIndex }
    ::= { msagCounterTable 1 }

    MsagCounterEntry ::=
    SEQUENCE {
    msagCounterIndex INTEGER,
    msagCounterName  DisplayString,
    msagCounterValue Counter64
    }

    -- 1.3.6.1.4.1.26757.2.4.1.1
    msagCounterIndex OBJECT-TYPE
    SYNTAX INTEGER(0..32768)
    MAX-ACCESS read-only
    STATUS current
    DESCRIPTION
    "MSAG Counter Index"
    ::= { msagCounterEntry 1 }

    -- 1.3.6.1.4.1.26757.2.4.1.2
    msagCounterName OBJECT-TYPE
    SYNTAX DisplayString (SIZE (0..48))
    MAX-ACCESS  read-only
    STATUS  current
    DESCRIPTION
    "MSAG Counter Name."
    ::= { msagCounterEntry 2 }

    -- 1.3.6.1.4.1.26757.2.4.1.3
    msagCounterValue OBJECT-TYPE
    SYNTAX Counter64
    MAX-ACCESS read-only
    STATUS  current
    DESCRIPTION
    "MSAG Counter Value."
    ::= { msagCounterEntry 3 }


-- trap section

    -- 1.3.6.1.4.1.26757.2.5
    alertMessage OBJECT-TYPE
    SYNTAX DisplayString
    MAX-ACCESS accessible-for-notify
    STATUS current
    DESCRIPTION
    "The string object containing textual description of alert."
    ::= { msag 5 }

    -- 1.3.6.1.4.1.26757.2.7
    alertSeverity OBJECT-TYPE
    SYNTAX  INTEGER {
    normal(1),
    warning(2),
    minor(3),
    major(4),
    critical(5)
    }
    -- MAX-ACCESS  read-write
    MAX-ACCESS not-accessible
    STATUS      current
    DESCRIPTION "Severity of alert"
    ::= { msag 7 }

    -- 1.3.6.1.4.1.26757.2.8
    alertId OBJECT-TYPE
    SYNTAX  DisplayString
    -- MAX-ACCESS  read-write
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION "Identifier of alert (e.g. object id)"
    ::= { msag 8 }

    -- 1.3.6.1.4.1.26757.2.9
    alertObjCategory OBJECT-TYPE
    SYNTAX  DisplayString
    -- MAX-ACCESS  read-write
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION "Object category."
    ::= { msag 9 }
